function renderFlow(steps) {
  if (!steps || steps.length === 0) return '';
  let html = '<div class="flow-row">';
  steps.forEach((s, i) => {
    html += `<span class="flow-item">${s}</span>`;
    if (i < steps.length - 1) html += '<span class="flow-arrow">→</span>';
  });
  html += '</div>';
  return html;
}

  if (!email) return '';
  return `
    <div class="email-box">
      <div class="email-header">✉️ Outlook 메일 작성 예시</div>
      <div class="email-meta">
        <div class="email-row"><span class="email-label">받는 사람</span><span>${email.to}</span></div>
        ${email.cc ? `<div class="email-row"><span class="email-label">참조</span><span>${email.cc}</span></div>` : ''}
        <div class="email-row"><span class="email-label">제목</span><span>${email.subject}</span></div>
      </div>
      <div class="email-body">${email.body}</div>
      ${email.attach ? `<div class="email-attach"><span class="attach-tag">첨부</span>${email.attach}</div>` : ''}
    </div>`;
}

function renderInfoGrid(cards) {
  if (!cards) return '';
  let html = '<div class="info-grid">';
  cards.forEach(c => {
    html += `<div class="info-card">
      <div class="info-card-title">${c.title}</div>
      <ul>${c.items.map(i => `<li>${i}</li>`).join('')}</ul>
    </div>`;
  });
  html += '</div>';
  return html;
}

function renderTable(table) {
  if (!table) return '';
  return `<table class="ptable">
    <thead><tr>${table.headers.map(h => `<th>${h}</th>`).join('')}</tr></thead>
    <tbody>${table.rows.map((r, i) => `<tr>${r.map(c => `<td>${c}</td>`).join('')}</tr>`).join('')}</tbody>
  </table>`;
}

let expandCount = 0;

function renderContent(key) {
  const d = DATA[key];
  if (!d) return '';
  let html = '';

  // Flow
  html += renderFlow(d.flow);

  // Steps
  if (d.steps && d.steps.length > 0) {
    d.steps.forEach(s => {
      const eid = 'exp_' + (++expandCount);
      html += `<div class="step">
        <div class="step-num">${s.n}</div>
        <div class="step-body">
          <div class="step-title">${s.title}</div>
          ${s.desc ? `<div class="step-desc">${s.desc}</div>` : ''}
          ${s.detail ? `<button class="expand-btn" onclick="toggleDetail('${eid}', this)">▶ 상세 보기</button><div class="expand-detail" id="${eid}">${s.detail}</div>` : ''}
          ${s.email ? renderEmail(s.email) : ''}
        </div>
      </div>`;
    });
  }

  if (d.infoGrid) { html += '<hr class="sec-div">'; html += renderInfoGrid(d.infoGrid); }
  if (d.table) { html += '<hr class="sec-div">'; html += renderTable(d.table); }
  if (d.warn) { html += `<div class="warn"><span>⚠️</span><div>${d.warn}</div></div>`; }
  if (d.note) { html += `<div class="note">${d.note}</div>`; }

  return html;
}

function toggleDetail(id, btn) {
  const el = document.getElementById(id);
  if (el.classList.contains('open')) {
    el.classList.remove('open');
    btn.textContent = '▶ 상세 보기';
  } else {
    el.classList.add('open');
    btn.textContent = '▲ 닫기';
  }
}

// ── Bubble selection ─────────────────────────────────────────────────────────

function selectBubble(el) {
  // Deactivate all
  document.querySelectorAll('.bubble').forEach(b => b.classList.remove('active'));
  el.classList.add('active');

  const key = el.dataset.key;
  const d = DATA[key];
  if (!d) return;

  // Update monitor
  document.getElementById('monitorIcon').textContent = d.icon;
  document.getElementById('monitorLabel').textContent = d.label;
  document.getElementById('monitorBadge').textContent = d.badge;

  const body = document.getElementById('monitorBody');
  body.style.opacity = '0';
  body.style.transform = 'translateY(6px)';

  setTimeout(() => {
    body.innerHTML = renderContent(key);
    body.style.transition = 'opacity 0.25s, transform 0.25s';
    body.style.opacity = '1';
    body.style.transform = 'translateY(0)';
    body.scrollTop = 0;
  }, 150);

  const wrap = document.getElementById('monitorWrap');
  wrap.classList.remove('hidden');
  wrap.classList.add('visible');
}

// ── Init: show first bubble ─────────────────────────────────────────────────
window.addEventListener('DOMContentLoaded', () => {
  const first = document.querySelector('.bubble.active');
  if (first) selectBubble(first);
});
